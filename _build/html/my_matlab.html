GetFEMMatmodel.m#

 1% Get the FEM matrices for the PDE model
 2function [model,FEM_M,FEM_K,FEM_F] = GetFEMMatmodel(xc,yc,model)
 3global EID
 4
 5% Heat source function
 6f = @(location,state)10000*exp(-((location.x - xc).^2+(location.y-yc).^2)/0.05);
 7
 8% Coefficients for the PDE
 9specifyCoefficients(model,'m',0,'d',1,'c',1,'a',0,'f',f);
10
11% Get the FEM matrices from the model
12model_FEM_matrices = assembleFEMatrices(model);
13FEM_M = model_FEM_matrices.M; % Mass matrix
14FEM_K = model_FEM_matrices.K; % Stiffness matrix
15FEM_F = model_FEM_matrices.F; % Load vector
16
17% Enforce boundary conditions
18FEM_M(EID,:)= 0 ;
19FEM_M(:,EID)= 0 ;
20FEM_M(EID,EID)= eye(length(EID));

SolveFOM.m#

 1% Solves FOM
 2function [t,T]=SolveFOM(u0,t,xc,yc)
 3global model FEM_M FEM_K FEM_F
 4
 5odesolve_tol = 1e-6; % Set the tolerance level
 6
 7% Configure the options for the ODE solver
 8options = odeset('Mass',FEM_M ,'AbsTol',odesolve_tol ,'RelTol',odesolve_tol ,'Stats','on');
 9disp('Solving FOM');
10tic
11[t , T] = ode45(@FOM_rhs,t,u0',options ); % Solve the ODEs
12toc
13T = T'; % Tranpose so each column is a solution at a point in time

FOM_rhs.m#

1% Defines the RHS for the ODEs
2function Yout = FOM_rhs(t,Y)
3global  FEM_K  FEM_F EID
4Yout = -(FEM_K*Y) + FEM_F; % Store the linear combination
5Yout(EID') = 0; % Apply boundary conditions
6end

ROM_rhs.m#

3global Kr fr

5end

ExtractCylinderT.m#
